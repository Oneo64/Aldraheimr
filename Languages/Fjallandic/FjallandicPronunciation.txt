	Í: /i/

	O: /ɔ/
	OR: /or/
	Ó: /ou/
	Œ: /oi/

	U: /ə/
	Ú: /u/

	Y: /ju/
	J: /j/

	AU: /au/
	ÁI: /ɑi/
	EI, EY: /ei/
	ÍU: /ju/
	ÚA: /u:ɑ/
	ÚI: /u:ɪ/

	DR: /dr/
	TR: /tr/

	KV, GV: /kf/, /gf/
	HV: /ʍ/
	GE, GI, GÍ: If in a VCV word structure, G is pronounced /dʒ/. Example: ægi is pronounced /'ai:dʒɪ/.
	ING: /ɪŋkʰ/

	Þ: /θ/
	Ð: /ð/

	T: /tʰ/

	R: When the R is at the end of the word or the syllable, it is slightly rolled. /r/
