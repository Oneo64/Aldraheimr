[[PRONUNCATION]]

[[MONOPHTHONGS]]
	A: /ɑ/
	Á: /ɑuː/
	Æ: /aiː/
	E: /ɛ/
	É: /jɛː/
	I: /ɪ/
	Í: /i/
	O: /o/ or /ɔ/, more commonly /o/
	Ó: /ouː/
	Ö: /œ/
	U: /ʌ/
	Ú: /u/
	Y: /y/
	Ý: /yiː/, or /yː/ if the next IPA is /j/

[[DIPHTHONGS]]
	AU: /œi/
	ÁI: /ɑi/
	EI, EY: /ɛi/
	ÍU: /iːju/
	ÚA: /uːɑ/
	ÚI: /ui/

[[CONSONANTS]]
	B: /b/
	D: /d/
	Ð: /ð/
	F: /f/
	G: /g/
	G in VCV structures, and the second V is E, I, Í, U, or Ú: /j/
		Example: ægi is pronounced /'ai.jɪ/, but getr is pronounced /'gɛtr/
	H: /h/
	J: /j/
	K: /k/
	L: /l/
	-L: /ɫ/
	-ULL, -UL: /ɫ/
	M: /m/
	N: /n/
	P: /p/
	R: /r/
	S: /s/
	T: /t/
	V: /v/
	HV: /ʍ/
	KV: /kf/
	X: /ks/
	Z: /z/
	Þ: /θ/

	-NG: /ŋk/

	Stress: Defaulted to the first syllable of the word.
